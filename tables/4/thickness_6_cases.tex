%usage: \LabGrid[colour]{CAPITAL letter for last row/column}
\newcommand\Grid[2][purple]{%
    \begin{tikzpicture}[xscale=1]
        \foreach \row [count=\r,remember=\r] in {6,...,#2} {
           %\draw[fill=#1!10!white](0,-\r) rectangle ++(2,1);
           \node at (1,0.5-\r){$\row$};
           \foreach \c in {1,...,\r} {
              \draw[gray, fill=gray!30](2*\c,-\r) rectangle ++(2,1);
           }
        }
        %\draw[fill=#1!10!white](0,-1-\r) rectangle ++(2,1);
        \foreach \row [count=\c] in {6,...,#2} {
            %\draw[fill=#1!10!white](2*\c,-1-\r) rectangle ++(2,1);
            \node at (2*\c+1,-0.5-\r) {\row};
        }
        \foreach \col in {2,8,14,20,26,32} {
       		\foreach \row in {0, ..., 21} {	  		
			\ifnum \numexpr-\col/2-\row > -14
				\draw[gray, fill=white](\col,-\col/2-\row) rectangle ++(2,1);
			\fi
        		}
        }
        \foreach \row in {-4, -7, ..., -13} {
       		\foreach \col in {2, 4, ..., 26} {	  		
			\ifnum \numexpr -\col/2 > \row
				\draw[gray, fill=white](\col, \row) rectangle ++(2,1);
			\fi
        		}
        }
    
        % 0 mod 6 by even
        \foreach \col in {2,14,26} {
       		\foreach \row in {0,2, ..., 20} {	  		
			\ifnum \numexpr-\col/2-\row > -14
				\draw[gray, fill=blue!30](\col,-\col/2-\row) rectangle ++(2,1);
				\node at (\col+1, -\col/2-\row+0.5) {$1$};
			\fi
        		}
        }
        \foreach \row in {-7, -13} {
       		\foreach \col in {2, 6, ..., 26} {	  		
			\ifnum \numexpr -\col/2 > \row
				\draw[gray, fill=blue!30](\col, \row) rectangle ++(2,1);
				\node at (\col+1, \row+0.5) {$1$};
			\fi
        		}
        }
        % 3 mod 6 by odd
       \foreach \col in {8,20,32} {
       		\foreach \row in {0,2, ..., 20} {	  		
			\ifnum \numexpr-\col/2-\row > -14
				\draw[gray, fill=green!30](\col,-\col/2-\row) rectangle ++(2,1);
				\node at (\col+1, -\col/2-\row+0.5) {$2$};
			\fi
        		}
        }
        \foreach \row in {-4, -10} {
       		\foreach \col in {4, 8, ..., 24} {	  		
			\ifnum \numexpr -\col/2 > \row
				\draw[gray, fill=green!30](\col, \row) rectangle ++(2,1);
				\node at (\col+1, \row+0.5) {$2$};
			\fi
        		}
        }
        % 3 mod 6 by even
        \foreach \col in {8,20,32} {
       		\foreach \row in {1,3, ..., 21} {	  		
			\ifnum \numexpr-\col/2-\row > -14
				\draw[gray, fill=red!30](\col,-\col/2-\row) rectangle ++(2,1);
				\node at (\col+1, -\col/2-\row+0.5) {$3$};
			\fi
        		}
        }
        \foreach \row in {-4, -10} {
       		\foreach \col in {2, 6, ..., 22} {	  		
			\ifnum \numexpr -\col/2 > \row
				\draw[gray, fill=red!30](\col, \row) rectangle ++(2,1);
				\node at (\col+1, \row+0.5) {$3$};
			\fi
        		}
        }
        
        % 0 mod 6 by odd
        \foreach \col in {2,14,26} {
       		\foreach \row in {1,3, ..., 21} {	  		
			\ifnum \numexpr-\col/2-\row > -14
				\draw[gray, fill=yellow!30](\col,-\col/2-\row) rectangle ++(2,1);
				\node at (\col+1, -\col/2-\row+0.5) {$4$};
			\fi
        		}
        }
        \foreach \row in {-7, -13} {
       		\foreach \col in {4, 8, ..., 24} {	  		
			\ifnum \numexpr -\col/2 > \row
				\draw[gray, fill=yellow!30](\col, \row) rectangle ++(2,1);
				\node at (\col+1, \row+0.5) {$4$};
			\fi
        		}
        }
        
        \draw[gray,fill=red!30] (2,-4) -- (4,-3)--(4,-4)--cycle;
        \draw[gray,fill=yellow!30] (2,-3)--(2,-4) -- (4,-3)--cycle;
        
        \draw[gray,fill=red!30] (2,-10) -- (4,-9)--(4,-10)--cycle;
        \draw[gray,fill=yellow!30] (2,-9)--(2,-10) -- (4,-9)--cycle;
        
        \draw[gray, fill=red!30] (8,-7) -- (10,-6)--(10,-7)--cycle;
        \draw[gray, fill=yellow!30] (8,-6)--(8,-7) -- (10,-6)--cycle;
        
        \draw[gray,fill=red!30] (8,-13) -- (10,-12)--(10,-13)--cycle;
        \draw[gray,fill=yellow!30] (8,-12)--(8,-13) -- (10,-12)--cycle;
        
        \draw[gray,fill=red!30] (14,-10) -- (16,-9)--(16,-10)--cycle;
        \draw[gray,fill=yellow!30] (14,-9)--(14,-10) -- (16,-9)--cycle;
        
        \draw[gray,fill=red!30] (20,-13) -- (22,-12)--(22,-13)--cycle;
        \draw[gray,fill=yellow!30] (20,-12)--(20,-13) -- (22,-12)--cycle;
    \end{tikzpicture}%
}
